    valorlogico    CHAR(100),

);

    id          INTEGER NOT NULL,
    nombre      CHAR(100),
    nota        INTEGER,

    id       INTEGER NOT NULL,
    nombre   CHAR(100)
);

ALTER TABLE cursos ADD CONSTRAINT cursos_pk PRIMARY KEY ( id );

    enunciado   CHAR(100),

--2. Conocer los cursos sin evaluaciones
select DISTINCT cursos.nombre from cursos full join unidades on cursos.id=unidades.cursos_id full join test on unidades.id=test.unidades_id where test.id is null;
--3  Determinar las evaluaciones con deficiencia. Una evaluación es deficiente:
--a. Si no tiene preguntas.
select test.nombre, count(test.nombre) from test full join preguntas on test.id=preguntas.test_id full join alternativas on preguntas.id=alternativas.preguntas_id group by test.nombre having count(preguntas.id)<=0;
--b. Si hay preguntas con 2 ó menos alternativas
select test.nombre, count(test.nombre) from test full join preguntas on test.id=preguntas.test_id full join alternativas on preguntas.id=alternativas.preguntas_id group by test.nombre having count(alternativas.id)<=2;
--c. Si todas las alternativas son correctas o si todas las alternativas son incorrectas.--;
select test.id from test where (select count (*) from test, preguntas, alternativas where test.id=preguntas.test_id and preguntas.id=alternativas.preguntas_id and alternativas.valorlogico='Y') =5;
select test.id from test where (select count (*) from test, preguntas, alternativas where test.id=preguntas.test_id and preguntas.id=alternativas.preguntas_id and alternativas.valorlogico='N') =5;

--4. Determinar cuántos alumnos hay en cada curso.
select cursos.nombre, count(alumnos.id) from  alumnos,cursos where cursos.id=alumnos.cursos_id group by cursos.nombre;
--5. Obtener el puntaje no normalizado de cada evaluación. El puntaje no normalizado ha sido definido (requerimiento) como: P = buenas – malas/4. Si un alumno no contesta en una pregunta exactamente lo mismo que se ha definido como correcto, la pregunta cuenta como mala a menos que el alumno haya omitido. 
select buenas, malas,omitidas,malas/4,buenas-(malas/4)  from testxalumno;
--6. Obtener el puntaje normalizado, o sea, de 1,0 a 7,0.
select buenas, malas,omitidas,malas/4,buenas-(malas/4) as Sin_Normalizar,((((buenas-malas)/4)+1)*2) as Normalizadas  from testxalumno;
--7. Nombre de estudiantes de un curso determinado que aprueban una evaluación determinada (donde la nota de aprobación mínima es un 4,0).
select substr(cursos.nombre,1,50) As Nombre_Del_curso, substr(alumnos.nombre,1,20)as Nombre_del_Alumno, substr(test.nombre,1,15) as Nombre_de_la_prueba,((((buenas-malas)/4)+1)*2) as Nota from alumnos,testxalumno,cursos,test where alumnos.id=testxalumno.alumnos_id and ((((buenas-malas)/4)+1)*2)>=4 and cursos.id=1 and test.id=1;
--8. Nota promedio de los estudiantes de un curso determinado, para una evaluación de terminada.
